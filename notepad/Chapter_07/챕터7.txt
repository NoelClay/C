#include <stdio.h>

int main() {
    int flag = 1;
    while(flag == 1) {
        printf("숫자 입력해봐 그 숫자만큼 *찍을거야 그만두고 싶으면 x를 입력해 : ");
        char input[100];  // 문자열 입력을 받기 위한 배열
        scanf("%s", input);  // 문자열로 입력 받음

        if(input[0] == 'x' && input[1] == '\0') {  // 'x'만 입력되었는지 확인
            flag = 0;
        } else {
            int nInput = atoi(input);  // 문자열을 정수로 변환
            if(nInput < 1)
                nInput = 1;
            else if(nInput > 9)
                nInput = 9;

            for(int i = 0; i < nInput; i++)
                putchar('*');
            puts("");
        }
    }
    printf("\n 종료합니다..\n");

    return 0;
}

#include <stdio.h>

int main() {
    int i=0, j=0;
    while(i<5) {
        while(j<5){
            printf("*\t");
            j++;
        }
        i++;
        j=0;
        printf("\n");
    }
    printf("\n 종료합니다..\n");

    return 0;
}

#include <stdio.h>

int main() {
    int nInput = 2, i = 1;
    while(nInput<=9 && nInput>=2) {
        puts("구구단 한단만 출력하기! 원하는 구구단 입력! 2보다 작고 9보다 크면 에러!");
        scanf("%d", &nInput);
        if(nInput>9 || nInput<2) {
            puts("Error");
            break;
        }
        else{
            while(i<=9){
                printf("%d * %d = %d\n", nInput, i, nInput*i);
                i++;
            }
        }
        i=0;
    }
    return 0;
}

역직각삼각형
#include <stdio.h>

int main() {
    for(int i=0; i<5 ; i++){
        for(int j = 1;j<5-i;j++){
            printf(" \t");
        }
        for(int k = 0; k<i+1; k++){
            printf("*\t");
        }
        puts("");
    }
    return 0;
}
